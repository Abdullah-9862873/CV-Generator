4. Database Schema (Prisma)
model User {
  id        String   @id @default(uuid())
  email     String   @unique
  cvs       CV[]
  createdAt DateTime @default(now())
}
model CV {
  id            String   @id @default(uuid())
  userId        String
  user          User     @relation(fields: [userId], references: [id])
  
  // Original upload
  originalImage String   // URL to stored image
  
  // Extracted data
  title         String?
  rawText       String   @db.Text
  structure     Json     // Parsed sections & layout
  
  // Editable content
  content       Json     // Full editable CV data
  
  // Settings
  template      String   @default("original")
  pageSize      String   @default("A4")
  
  // Metadata
  createdAt     DateTime @default(now())
  updatedAt     DateTime @updatedAt
  exports       Export[]
}
model Export {
  id          String   @id @default(uuid())
  cvId        String
  cv          CV       @relation(fields: [cvId], references: [id])
  format      String   // pdf, docx, html
  downloadUrl String
  createdAt   DateTime @default(now())
}

// 1. OCR - Using Tesseract.js (FREE, runs in browser)
import Tesseract from 'tesseract.js';
const extractText = async (imageFile) => {
  const result = await Tesseract.recognize(
    imageFile,
    'eng',
    { logger: m => console.log(m) }
  );
  return result.data;
};
// 2. Layout Analysis - Using Ollama (FREE, local)
const analyzeLayout = async (imageBase64) => {
  const response = await fetch('http://localhost:11434/api/generate', {
    method: 'POST',
    body: JSON.stringify({
      model: 'llama3.2-vision',
      prompt: `Analyze this CV image and extract structure as JSON:
      {
        "sections": ["header", "experience", "education", "skills"],
        "layout": "single-column" | "two-column",
        "fonts": ["font-names"],
        "colors": ["color-hex-codes"]
      }`,
      images: [imageBase64]
    })
  });
  return response.json();
};
// 3. Database - Supabase Free Tier
import { createClient } from '@supabase/supabase-js';
const supabase = createClient('your-url', 'your-key');
// 4. Storage - Supabase (1GB free)
const uploadImage = async (file) => {
  const { data, error } = await supabase
    .storage
    .from('cv-uploads')
    .upload(`${Date.now()}-${file.name}`, file);
  return data;
};
// 5. PDF Export - html2pdf.js (FREE, client-side)
import html2pdf from 'html2pdf.js';
const exportPDF = (element) => {
  html2pdf().from(element).save('cv.pdf');
};
